// Pin definitions const int sensorPin = 2; //(D2) - Sensor input const int ledControlPin = 23; //(D23) - Controls transistor void setup() { Serial.begin(115200); pinMode(sensorPin, INPUT); pinMode(ledControlPin, OUTPUT); digitalWrite(ledControlPin, LOW); } void loop() { int sensorValue = digitalRead(sensorPin); Serial.println(sensorValue); // Control LEDs based on sensor if (sensorValue == HIGH) { // Sensor is triggered – turn ON LEDs digitalWrite(ledControlPin, HIGH); } else { // Sensor not triggered – turn OFF LEDs digitalWrite(ledControlPin, LOW); } // Small delay to stabilize readings delay(100); }
